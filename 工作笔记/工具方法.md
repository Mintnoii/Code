## 判断是否为空的 JS 公共方法

```js
/**
 * 判断参数是否为空
 * @param {[type]} keys [description]
 */
var IsEmpty = function(keys){
    if (typeof(keys) === "string") {
        keys = keys.replace(/(^\s*)|(\s*$)/g, "");
        if (keys == "" || keys == null || keys == 'null' || keys == undefined || keys == 'undefined') {
            return true
        } else {
            return false
        }
    } else if (typeof(keys) === 'undefined') {
        return true;
    } else {
        if (typeof(keys) == "object") {
            for(let i in keys){
                return false;
            }
            return true;
        }
    }
};
export {IsEmpty};
```

## 时间格式

```js
export function dateTimeFormatter(cell) {
    let date = new Date(cell);
    let m = date.getMonth() + 1;
    m = m < 10 ? ('0' + m) : m;
    let d = date.getDate();
    d = d < 10 ? ('0' + d) : d;
    let h = date.getHours();
    let minute = date.getMinutes();
    let second = date.getSeconds();
    minute = minute < 10 ? ('0' + minute) : minute;
    second = second < 10 ? ('0' + second) : second;
    return m + '-' + d+' '+h+':'+minute+ ':' + second;
}

export function dateFormatter(cell) {
    let date = new Date(cell);
    var y = date.getFullYear();
    var m = date.getMonth() + 1;
    m = m < 10 ? ('0' + m) : m;
    var d = date.getDate();
    d = d < 10 ? ('0' + d) : d; 
    return y + '-' + m + '-' + d;
}

export function timeFormatter(cell) {
    let date = new Date(cell);
    let h = date.getHours();
    let minute = date.getMinutes();
    minute = minute < 10 ? ('0' + minute) : minute;
    return h+':'+minute;
}
```

## 路由参数

```js
/**
   * 获取url中指定参数
   * @param  {[type]} name [key]
   * @return {[type]}      [description]
   */
getUrlParam: function(name) {
    var reg = new RegExp('(^|&)' + name + '=([^&]*)(&|$)');
    var r = decodeURIComponent(window.location.search.substr(1)).match(reg);
    if (r != null) return unescape(r[2]);
    return null;
}
```
